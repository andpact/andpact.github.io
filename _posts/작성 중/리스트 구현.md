# **List 구현**
- **List Interface**
```java
public interface ListInterface<E> {
    public void add(int i, E x);
    public void append(E x);
    public E remove(int i);
    public boolean removeItem(E x);
    public E get(int i);
    public void set(int i, E x);
    public int indexOf(E x);
    public int len();
    public boolean isEmpty();
    public void clear();
}
```

```java
public class ArrayList<E> implements ListInterface<E> {
    private E item[];        
    private int numItems;
    private static final int DEFAULT_CAPACITY = 64;

    // 생성자1
    public ArrayList() {
        item = (E[]) new Object[DEFAULT_CAPACITY];
        numItems = 0;
    }

    // 생성자2
    public ArrayList(int n) {
        item = (E[]) new Object[n];
        numItems = 0;
    }	
	
    // k번째 자리에 요소 x 삽입
    public void add(int index, E x) {
        if (item.length <= numItems || index < 0 || numItems < index) { /* 에러 처리 */ }
        else {
            for (int i = numItems - 1; index <= i; i--)
                item[i + 1] = item[i]; // 오른쪽 시프트
            item[index] = x;
            numItems++;
        }
    }
    
    // 맨 뒤에 요소 추가
    public void append(E x) {
        if (item.length <= numItems) { /* 에러 처리 */ } 
        else item[numItems++] = x;
    }
    
    // k번째 요소 삭제
    public E remove(int index) {
        if (isEmpty() || index < 0 || numItems - 1 < index) return null;
        else {
            E tmp = item[index];
            for (int i = index; i <= numItems - 2; i++)
                item[i] = item[i + 1];
            numItems--;
            return tmp;
        }
    }

    // 요소 x 삭제
    public boolean removeItem(E x) {
        int k = 0;
        while (k < numItems && ((Comparable)item[k]).compareTo(x) != 0) k++;
        if (k == numItems) return false;
        else {
            for (int i = k; i <= numItems - 2; i++)
                item[i] = item[i+1];
            numItems--;
            return true;
        }
    }

    // i번째 요소 조회
    public E get(int index) { // 0-indexed
        if (0 <= index && index <= numItems - 1) return item[index]; 
        else return null;
    }
  
    // i번째 요소를 x로 변경
    public void set(int index, E x) {
        if (0 <= index && index <= numItems - 1) item[index] = x; 
        else { /* 에러 처리 */ }
    }
    
    // 요소 x의 인덱스 조회
    private final int NOT_FOUND = -12345;
    public int indexOf(E x) {
        int i = 0;
        for (i = 0; i < numItems; i++)
            if (((Comparable)item[i]).compareTo(x) == 0) return i;
        return NOT_FOUND;
    }
  
    // 총 요소 수 조회
    public int len() {
        return numItems;
    }

    // 비었는지 조회
    public boolean isEmpty() { 
        return numItems == 0;
    }

    // 전체 요소 삭제
    public void clear() {
        item = (E[]) new Object[item.length]; 
        numItems = 0;
    }

    // 전체 요소 출력
    public void printAll() {
        System.out.print("Print list (#items=" + numItems + ") ");
        for(int i = 0; i < numItems; i++) 
            System.out.print(item[i] + " ");
        System.out.println();
    }
}
```

```java
public class LinkedList<E> implements ListInterface<E> {
    private Node<E> head;
    private int numItems;

    public LinkedList() { // 생성자
        numItems = 0;
        head = new Node<>(null, null); 	// 더미 헤드 노드
    }

    // k번째 자리에 노드 x 삽입
    public void add(int index, E item) {
        if (0 < = index && index <= numItems) {
            Node<E> prevNode = getNode(index - 1);
            Node<E> newNode = new Node<>(item, prevNode.next);
            prevNode.next = newNode;
            numItems++;
        }
    }
  
    // 마지막에 노드 x 추가
    public void append(E item) {
        Node<E> prevNode = head; // 더미 노드
        while (prevNode.next != null)
            prevNode = prevNode.next;
        Node<E> newNode = new Node<>(item, null);
        prevNode.next = newNode;
        numItems++;
    }

    // k번째 노드 삭제
    public E remove(int index) {
        if (0 <= index && index < numItems) {
            Node<E> prevNode = getNode(index - 1);
            Node<E> currNode = prevNode.next;
            prevNode.next = currNode.next;
            numItems--;
            return currNode.item;
        } else
            return null;
    }

    // 노드 x 삭제
    public boolean removeItem(E x) {
        Node<E> currNode = head; // 더미 헤드
        Node<E> prevNode;
        for (int i = 0; i < numItems; i++) {
            prevNode = currNode;
            currNode = currNode.next;
            if (((Comparable)(currNode.item)).compareTo(x) == 0) {
                prevNode.next = currNode.next;
                numItems--;
                return true;
            }
        }
        return false;
    }

    // k번째 노드의 값 조회
    public E get(int index) {
        if (0 <= index && index <= numItems - 1) {
            return getNode(index).item;
        } else
            return null;
    }
  
    // k번째 노드 조회
    public Node<E> getNode(int index) {
        if (-1 <= index && index <= numItems - 1) {
            Node<E> currNode = head; // 더미 노드
            for (int i = 0; i <= index; i++)
                currNode = currNode.next;
            return currNode;
        } else
            return null;
    }

    // 연결 리스트의 k번째 노드를 x로 변경
    public void set(int index, E x) {
        if (0 <= index && index <= numItems - 1) getNode(index).item = x;
        else { /* 에러처리 */ }
    }

    // x값을 가진 노드의 인덱스 조회
    public final int NOT_FOUND = -12345;
    public int indexOf(E x) {
        Node<E> currNode = head; // 더미 노드
        int i;
        for (i = 0; i < numItems; i++) {
            currNode = currNode.next;
            if (((Comparable)(currNode.item)).compareTo(x) == 0)
                return i;
        }
        return NOT_FOUND;
    }
    
    // 총 요소 수 조회
    public int len() {
        return numItems;
    }
    
    // 비었는지 조회
    public boolean isEmpty() {
        return numItems == 0;
    }

    // 전체 노드 삭제
    public void clear() {
        numItems = 0;
        head = new Node<>(null, null);
    }
    
    // 전체 노드 출력
    public void printAll() {
        Node<E> t;
        System.out.print("Print list (#items=" + numItems + ") ");
        for (t = head.next; t != null; t = t.next) {
            System.out.print(t.item + " ");
        }
        System.out.println();
    }
}
```

```java
public class CircularLinkedList<E> implements ListInterface<E> {
	private Node<E> tail;
	private int numItems;

    // 생성자
	public CircularLinkedList() {
		numItems = 0;
		tail = new Node(-1);
		tail.next = tail;
	}

    // k번째 자리에 노드 x 삽입
	public void add(int index, E x) {
		if (0 <= index && index <= numItems) {
			Node<E> prevNode = getNode(index - 1);
			Node<E> newNode = new Node(x, prevNode.next);
			prevNode.next = newNode;
			if (index == numItems)
				tail = newNode;
			numItems++;
		}
	}

    // 마지막에 노드 x 삽입
	public void append(E x) {
		Node<E> prevNode = tail; // 더미 노드
		Node<E> newNode = new Node(x, tail.next);
		prevNode.next = newNode;
		tail = newNode;
		numItems++;
	}
    
    // k번째 노드 삭제
	public E remove(int index) {
		if (0 <= index && index <= numItems - 1) {
			Node<E> prevNode = getNode(index - 1);
			E rItem = prevNode.next.item;
			prevNode.next = prevNode.next.next;
			if (index == numItems)
				tail = prevNode;
			numItems--;
			return rItem;
		} else return null;
	}

    // 노드 x 삭제
	public boolean removeItem(E x) {	
		Node<E> currNode = tail.next; // 더미 헤드
		Node<E> prevNode;
		for (int i = 0; i < numItems; i++) {
			prevNode = currNode;
			currNode = currNode.next;
			if (((Comparable)(currNode.item)).compareTo(x) == 0) {
				prevNode.next = currNode.next;
				numItems--; 
				return true;
			}	
		}
		return false; // not found
	}

    // k번째 노드의 값 조회
	public E get(int index) {
		if (0 <= index && index <= numItems - 1) {
			return getNode(index).item; 
		} else return null;	// 에러
	}

    // k번째 노드 조회
	private Node<E> getNode(int index) {
		if (-1 <= index && index <= numItems) {
			Node<E> currNode = tail.next;  // 더미 헤드
			for (int i = 0; i <= index; i++)
				currNode = currNode.next;
			return currNode; 
		} else return null;
	}

    // k번째 노드의 값을 x로 변경
	public void set(int index, E x) {
		if (0 <= index && index <= numItems - 1) {
			getNode(index).item = x; 
		} else { /* 에러 처리 */ }
	}

    // x값을 가진 노드의 인덱스 조회
	public final int NOT_FOUND = -12345;
	public int indexOf(E x) {
		Node<E> currNode = tail.next; // 더미 헤드
		for (int i = 0; i <= numItems-1; i++) {
			currNode = currNode.next;
			if (((Comparable)(currNode.item)).compareTo(x) == 0) return i;
		}
		return NOT_FOUND; 
	}

    // 총 요소 수 조회
	public int len() {
		return numItems;
	}

    // 비었는지 조회
	public boolean isEmpty() { 
		return numItems == 0;
	}

    // 전체 노드 삭제
	public void clear() {
		numItems = 0;
		tail = new Node(-1);
		tail.next = tail;
	 }

	// 전체 노드 출력
	 public void printAll() {
		Node<E> head = tail.next; // 더미 헤드
		System.out.print("Print list (#items=" + numItems + ") ");
		for(Node<E> t = head.next; t != head; t = t.next)
			System.out.print(t.item + " ");
		System.out.println();
	}
}
```

```java
public class BidirectionalNode<E> {
	public BidirectionalNode<E> prev;
 	public E item;
 	public BidirectionalNode<E> next;
 	public BidirectionalNode() {
 		prev = next = null;
 		item = null;
 	}
 	public BidirectionalNode(E newItem) {
 		prev = next = null;
 		item = newItem;
 	}
 	public BidirectionalNode(BidirectionalNode<E> prevNode, E newItem, BidirectionalNode<E> nextNode) {
 		prev = prevNode; next = nextNode;
 		item = newItem;
 	}
}
```

```java
public class CircularDoublylLinkedList<E> implements ListInterface<E> {
	private BidirectionalNode<E> head;
	private int numItems;

    // 생성자
	public CircularDoublyLinkedList() {
		numItems = 0;
		head = new BidirectionalNode<>(null); // 더미 헤드
		head.next = head.prev = head;
	} 

    // k번째 자리에 노드 x 삽입
	public void add(int index, E x) {
		if (0 <= index && index <= numItems) {
			BidirectionalNode<E> prevNode = getNode(index - 1);
 			BidirectionalNode<E> newNode = new BidirectionalNode<>(prevNode, x, prevNode.next);
			newNode.next.prev = newNode;
			prevNode.next = newNode;
			numItems++;
		} else { /* 에러 처리 */ }
	}

    // 마지막에 노드 x 삽입
	public void append(E x) {
		BidirectionalNode<E> prevNode = head.prev;
		BidirectionalNode<E> newNode = new BidirectionalNode<>(prevNode, x, head);
		prevNode.next = newNode;
		head.prev = newNode;
		numItems++;
	}

    // k번째 노드 삭제
	public E remove(int index) {
		if (0 <= index && index <= numItems - 1) {
			BidirectionalNode<E> currNode = getNode(index);
			currNode.prev.next = currNode.next;
			currNode.next.prev = currNode.prev;
			numItems--;
			return currNode.item;
		} else
			return null;
	}
    
    // 노드 x 삭제
	public boolean removeItem(E x) {
		BidirectionalNode<E> currNode = head; // 더미 헤드
		for (int i = 0; i <= numItems - 1; i++) {
			currNode = currNode.next;
			if (((Comparable)(currNode.item)).compareTo(x) == 0) {
				currNode.prev.next = currNode.next;
				currNode.next.prev = currNode.prev;
 				numItems--;
				return true;
			}
		}
		return false;
	}

			return getNode(index).item;
		} else
			return null; // 에러
	}

				for (int i = 0; i <= index; i++)
					currNode = currNode.next;
			else
				for (int i = numItems - 1; index <= i; i--)
					currNode = currNode.prev;
			return currNode;
		} else
			return null; // 에러
	}

    // k번째 노드의 값을 x로 변경
	public void set(int index, E x) {
		if (0 <= index && index <= numItems - 1) {
			getNode(index).item = x;
		} else { /* 에러 처리 */ }

 	public int indexOf(E x) { 
		BidirectionalNode<E> currNode = head;

			currNode = currNode.next;
			if (((Comparable)(currNode.item)).compareTo(x) == 0)
				return i;
		}
```
