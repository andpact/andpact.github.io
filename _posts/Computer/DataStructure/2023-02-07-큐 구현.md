# **Queue**
- **Queue Interface**
```java
public interface QueueInterface<E> {
	public void enqueue(E x);
	public E dequeue();
	public E front();
	public boolean isEmpty();
	public void dequeueAll();
}
```
- **ArrayQueue Class**
```java
public class ArrayQueue<E> implements QueueInterface<E> {
	private E queue[];
	private int front, tail, numItems;
	private static final int DEFAULT_CAPACITY = 64;
	private final E ERROR = null;  // 임의의 에러 값

	// 생성자1
	public ArrayQueue() { 
		queue = (E[]) new Object[DEFAULT_CAPACITY];
		front = 0;
		tail = DEFAULT_CAPACITY - 1;
		numItems = 0;
	}

	// 생성자 2
	public ArrayQueue(int n) { 
		queue = (E[]) new Object[n];
		front = 0;
		tail = n - 1;
		numItems = 0;
	}

	// 요소 삽입
	public void enqueue(E newItem) {
		if (isFull()) System.out.println("Queue Full!");
		else {
			tail = (tail+1) % queue.length;
			queue[tail] = newItem;
			++numItems;
		}
	}

	// 맨 앞 요소 조회
	public E front() {
		if (isEmpty()) return ERROR;
		else return queue[front];
	}

	// 비었는지 조회
	public boolean isEmpty( ) {
		return (numItems == 0);
	}

	// 꽉 찼는지 조회
	public boolean isFull() {
		return (numItems == queue.length);
	}
 
	// 요소 삭제
	public E dequeue() {
		if (isEmpty()) return ERROR;
		else {
			E queueFront = queue[front];
			front = (front + 1) % queue.length;
			--numItems;
			return queueFront;
		}
	}

	// 전체 요소 삭제
	public void dequeueAll ( ) {
		queue = (E[]) new Object[queue.length];
		front = 0;
		tail = queue.length - 1;
		numItems = 0;
	}

	// 전체 요소 출력
	private void printAll() {
		int j = front;
		System.out.print("PrintAll: ");
		for(int i = 0; i < numItems; i++) {
			System.out.print(queue[j] + " ");
			j = (j + 1) % queue.length;
		}
		System.out.println();
	}
}
```
- **ListQueue Class**
```java
public class ListQueue<E> implements QueueInterface<E> {
	private ListInterface<E> list;

	// 생성자
	public ListQueue() {
		list = new ArrayList<E>();
	}

	// 요소 삽입
	public void enqueue(E newItem) {
		list.append(newItem);
	}
	
	// 맨 앞 요소 조회
	public int front(){
		return list.get(0);
	}

	// 꽉 찼는지 조회
	public boolean isEmpty() {
		return list.isEmpty();
	}
  
	// 요소 삭제
	public E dequeue() {
		return list.remove(0);
	}

	// 전체 요소 삭제
	public void dequeueAll() {
		list.clear();
	} 

	// 전체 요소 출력
	public void printAll() {
		list.printAll();
	}
}
```

- **LinkedQueue Class**
```java
public class LinkedQueue<E> implements QueueInterface<E>{
	private Node<E> tail;
	private final E ERROR = null; // 임의의 에러 값
	public LinkedQueue( ) {
		tail = null;
	}
 
	// 요소 삽입
	public void enqueue(E newItem) {
		Node<E> newNode = new Node<>(newItem);
		if (isEmpty( )) {
			newNode.next = newNode;
			tail = newNode;
		} else {
			newNode.next = tail.next;
			tail.next = newNode;
			tail = newNode;
		}
	}
		
	// 맨 앞 요소 조회
	public E front() {
		if (isEmpty()) return ERROR; 
		else return tail.next.item;	// tail.next: front
	}
		
	// 비었는지 조회
	public boolean isEmpty() {
		return (tail == null);
	}
		
	// 요소 삭제
	public E dequeue( ) {
		if (isEmpty( )) return ERROR;
		else {
			Node<E> front = tail.next;
			if (front == tail) tail = null; // 하나의 요소만 존재할 때
			else tail.next = front.next;
			return  front.item; 
		}
	}
		 
	// 전체 요소 삭제
	public void dequeueAll ( ) {
			tail = null;
	} 
	
	// 전체 요소 출력
	public void printAll() {
		Node<E> t = tail;
		System.out.println("Queue from front to tail");
		if (!isEmpty()) 
			do {
				t = t.next;
				System.out.println(t.item);
			} while (t != tail);
		else System.out.println("Empty Queue!");
	}
}
```
